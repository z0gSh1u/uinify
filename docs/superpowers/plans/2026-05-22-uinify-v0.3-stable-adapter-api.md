# Uinify v0.3 Stable Adapter API Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a small documented adapter authoring surface that returns canonical `UiStreamEvent[]` plus diagnostics, so consumers can build host-specific mappers against a stable contract instead of copying ad hoc examples.

**Architecture:** Add a lightweight adapter helper layer under `src/adapters/` that stays protocol-agnostic and terminates at canonical `UiStreamEvent[]`. Keep the runtime unchanged, expose stable diagnostics and validation helpers from the package root, and rewrite the reference mappers to use the new stable contract as recipes rather than as standalone patterns.

**Tech Stack:** TypeScript, Vitest, existing `createChatRuntime` runtime API, current example adapter fixtures and docs set

`src/adapters/validate-events.ts`

```ts
import type { UiStreamEvent } from "../runtime/events"
import type { UiAdapterDiagnostic, UiAdapterResult } from "./contracts"

export function createAdapterResult(
  events: UiStreamEvent[],
  diagnostics: UiAdapterDiagnostic[] = [],
): UiAdapterResult {
  return { events, diagnostics }
}

export function validateAdapterEvents(
  events: UiStreamEvent[],
  input: { knownMessageIds?: ReadonlySet<string> } = {},
): UiAdapterDiagnostic[] {
  const diagnostics: UiAdapterDiagnostic[] = []
  const seenMessageIds = new Set(input.knownMessageIds ?? [])

  if (events.length === 0) {
    diagnostics.push({
      level: "warning",
      code: "empty-output",
      message: "Adapter returned no canonical events for this input.",
    })
  }

  for (const event of events) {
    if (event.type === "message.started") {
      seenMessageIds.add(event.messageId)
      continue
    }

    if (
      (event.type === "message.completed" ||
        event.type === "message.failed" ||
        event.type.startsWith("part.")) &&
      !seenMessageIds.has(event.messageId)
    ) {
      diagnostics.push({
        level: "warning",
        code: "missing-message-start",
        message: `Received ${event.type} before message.started for message ${event.messageId}.`,
      })
    }

    if (event.type === "part.artifact.emitted" && event.artifact.views.length === 0) {
      diagnostics.push({
        level: "warning",
        code: "invalid-artifact",
        message: `Artifact ${event.artifact.id} must include at least one view.`,
      })
    }
  }

  return diagnostics
}
```

`src/adapters/run-adapter-step.ts`

```ts
import type { UiStreamEvent } from "../runtime/events"
import type { UiAdapterResult } from "./contracts"
import { createAdapterResult, validateAdapterEvents } from "./validate-events"

export function createAdapterRunner<TInput>(
  map: (input: TInput) => UiStreamEvent[],
): (input: TInput) => UiAdapterResult {
  const knownMessageIds = new Set<string>()

  return (input: TInput) => {
    const events = map(input)
    const diagnostics = validateAdapterEvents(events, { knownMessageIds })

    for (const event of events) {
      if (event.type === "message.started") {
        knownMessageIds.add(event.messageId)
      }
    }

    return createAdapterResult(events, diagnostics)
  }
}

export { createAdapterRunner }
```

import { createAdapterRunner, type UiStreamEvent } from "../../src"

export function mapOpenAiLikeChunk(chunk: OpenAiLikeChunk): UiStreamEvent[] {
  switch (chunk.type) {
    case "response.started":
      return [{ type: "message.started", messageId: chunk.response.response_id, role: chunk.response.speaker }]
    case "response.output_text.delta":
      return [{ type: "part.text.delta", messageId: chunk.response_id, partId: chunk.item.item_id, delta: chunk.delta }]
    case "response.tool.updated":
      return [{ type: "part.tool.updated", messageId: chunk.response_id, partId: chunk.tool_call.call_id, toolName: chunk.tool_call.name, status: mapToolPhase(chunk.tool_call.phase), inputSummary: chunk.tool_call.arguments_text ? `query: ${JSON.parse(chunk.tool_call.arguments_text).query as string}` : undefined, outputSummary: chunk.tool_call.result?.summary }]
    case "response.artifact":
      return [{ type: "part.artifact.emitted", messageId: chunk.response_id, partId: chunk.asset.artifact_id, artifact: mapArtifact(chunk.asset) }]
    case "response.completed":
      return [{ type: "message.completed", messageId: chunk.response.response_id }]
  }
}

function mapHostEvent(event: HostEvent): UiStreamEvent[] {

const adaptHostEvent = createAdapterRunner(mapHostEvent)

for (const result of incomingEvents.map(adaptHostEvent)) {
  if (result.diagnostics.length > 0) {
    console.warn(result.diagnostics)
  }

  for (const streamEvent of result.events) {
    runtime.dispatch(streamEvent)
  }
}
