# Uinify v0.2 Upload Lifecycle Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a host-driven attachment lifecycle that covers queued, uploading, uploaded, error, and removed states across composer and message surfaces without introducing an official upload client.

**Architecture:** Reuse one canonical attachment shape across composer and message rendering, but keep composer state host-controlled through props instead of moving file uploads into the runtime. Extend runtime events only for message-bound attachment parts, and keep send-policy enforcement inside the composer UI.

**Tech Stack:** TypeScript, React 18, Lexical, Vitest, Testing Library

---

```ts
export type LexicalComposerProps = {
  onSubmit: (value: UiComposerValue) => void
  initialAttachments?: UiComposerAttachment[]
  attachments?: UiComposerAttachment[]
  onAttachmentsChange?: (attachments: UiComposerAttachment[]) => void
  onAttachmentRetry?: (attachment: UiComposerAttachment) => void
  sendPolicy?: "allow-pending" | "uploaded-only"
  slashCommands?: UiComposerChoice[]
  mentions?: UiComposerChoice[]
}

export function LexicalComposer({
  attachments: controlledAttachments,
  initialAttachments = [],
  onAttachmentRetry,
  onAttachmentsChange,
  onSubmit,
  sendPolicy = "allow-pending",
  slashCommands = [],
  mentions = [],
}: LexicalComposerProps) {
const [uncontrolledAttachments, setUncontrolledAttachments] = useState(initialAttachments)
const attachments = controlledAttachments ?? uncontrolledAttachments

const updateAttachments = (next: UiComposerAttachment[] | ((current: UiComposerAttachment[]) => UiComposerAttachment[])) => {
  const resolved = typeof next === "function" ? next(attachments) : next
  setUncontrolledAttachments(resolved)
  onAttachmentsChange?.(resolved)
}

const handleRemove = (id: string) => {
  updateAttachments((current) =>
    current.map((attachment) =>
      attachment.id === id ? { ...attachment, status: "removed" } : attachment,
    ),
  )
}

const canSubmit =
  sendPolicy !== "uploaded-only" ||
  attachments.every((attachment) => attachment.status !== "queued" && attachment.status !== "uploading")

<AttachmentTray attachments={attachments} onRemove={handleRemove} onRetry={onAttachmentRetry} />

<button
  disabled={!canSubmit}
  onClick={() => {
    if (!canSubmit) {
      return
    }

    onSubmit({ text, attachments, commands: submitCommands, mentions: submitMentions })
  }}
  type="button"
>
  Send
</button>
}
```

```tsx
export type AttachmentTrayProps = {
  attachments: UiComposerAttachment[]

  onRetry?: (attachment: UiComposerAttachment) => void
}

{attachments.filter((attachment) => attachment.status !== "removed").map((attachment) => (
  <div key={attachment.id} data-slot="attachment-item" data-state={attachment.status}>
    <span>{attachment.name}</span>
    {attachment.progress !== undefined ? <span>{attachment.progress}% uploaded</span> : null}
    {attachment.error ? <span>{attachment.error}</span> : null}
    {attachment.status === "error" && onRetry ? (
      <button onClick={() => onRetry(attachment)} type="button">Retry</button>
    ) : null}
    {onRemove ? (
      <button onClick={() => onRemove(attachment.id)} type="button">Remove</button>
    ) : null}
  </div>
))}
```

```tsx
import type { UiAttachmentPart } from "../model/types"

export type AttachmentPartProps = {
  part: UiAttachmentPart
}

export function AttachmentPart({ part }: AttachmentPartProps) {
  const { attachment } = part

  return (
    <div data-slot="attachment-part" data-state={attachment.status}>
      {attachment.remoteUrl ? (
        <a href={attachment.remoteUrl}>{attachment.name}</a>
      ) : (
        <span>{attachment.name}</span>
      )}
      {attachment.progress !== undefined ? <span>{attachment.progress}%</span> : null}
      {attachment.error ? <span>{attachment.error}</span> : null}
    </div>
  )
}
```
