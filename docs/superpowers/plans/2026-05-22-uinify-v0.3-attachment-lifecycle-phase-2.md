`src/composer/lexical/plugins/attachment-plugin.tsx`

```ts
export function collectAttachments(files: FileList | File[]) {
  return Array.from(files).map<UiComposerAttachment>((file, index) => ({
    id: `${Date.now()}-${attachmentIdCounter++}-${index}-${file.name}`,
    file,
    name: file.name,
    mimeType: file.type,
    size: file.size,
    status: "queued",
    sourceAttachmentId: `${Date.now()}-${index}-${file.name}`,
  }))
}

export function createAttachmentHandlers(
  onAdd: (files: UiComposerAttachment[]) => void,
  onValidate?: (attachments: UiComposerAttachment[]) => UiAttachmentValidationResult,
) {
  function accept(files: FileList | File[]) {
    const next = collectAttachments(files)
    const validated = onValidate ? onValidate(next) : { accepted: next, rejected: [] }
    onAdd(validated.accepted)
    return validated.rejected
  }

  return {
    onPaste(event: ClipboardEvent<HTMLElement>) {
      if (event.clipboardData.files.length > 0) {
        event.preventDefault()
        return accept(event.clipboardData.files)
      }

      return []
    },
    onDrop(event: DragEvent<HTMLElement>) {
      if (event.dataTransfer.files.length > 0) {
        event.preventDefault()
        return accept(event.dataTransfer.files)
      }

      return []
    },
  }
}
```

`src/composer/lexical/lexical-composer.tsx`

```tsx
export type LexicalComposerProps = {
  onSubmit: (value: UiComposerValue) => void
  onAttachmentValidation?: (attachments: UiComposerAttachment[]) => UiAttachmentValidationResult
  onAttachmentCancel?: (attachment: UiComposerAttachment) => void
}

const [rejectedAttachments, setRejectedAttachments] = useState<UiComposerAttachmentRejection[]>([])

const attachmentHandlers = useMemo(
  () =>
    createAttachmentHandlers(
      (items) => updateAttachments((current) => [...current, ...items]),
      (items) => {
        const result = onAttachmentValidation ? onAttachmentValidation(items) : { accepted: items, rejected: [] }
        setRejectedAttachments(result.rejected)
        return result
      },
    ),
  [onAttachmentValidation],
)

const submittableAttachments = attachments.filter((attachment) => attachment.status !== "removed")
const isSubmitBlocked =
  sendPolicy === "uploaded-only" &&
  submittableAttachments.some((attachment) => attachment.status !== "uploaded")
const sendBlockedReason = isSubmitBlocked
  ? "Waiting for upload to complete before send."
  : undefined

<AttachmentTray
  attachments={attachments.map((attachment) =>
    attachment.status === "uploaded" || !sendBlockedReason
      ? attachment
      : { ...attachment, blockedReason: sendBlockedReason },
  )}
  rejectedAttachments={rejectedAttachments}
  onCancel={onAttachmentCancel}
  onRemove={(id) => {
    updateAttachments((items) => items.filter((item) => item.id !== id))
  }}
  onRetry={onAttachmentRetry}
/>
```

export type AttachmentTrayProps = {
  attachments: UiComposerAttachment[]
  rejectedAttachments?: UiComposerAttachmentRejection[]

  onRetry?: (attachment: UiComposerAttachment) => void
  onCancel?: (attachment: UiComposerAttachment) => void

function getAttachmentStageLabel(status: UiComposerAttachment["status"]) {
  switch (status) {
    case "queued":
      return "Queued"
    case "uploading":
      return "Uploading"
    case "uploaded":
      return "Uploaded"
    case "error":
      return "Upload failed"
    case "removed":
      return "Removed"
  }
}

{attachment.status === "uploading" && onCancel ? (

    Cancel upload
  </button>
) : null}

`docs/integration/upload-lifecycle.md`

```md
## Validation, Rejection, And Cancel Intent

Use `onAttachmentValidation` when the host needs to reject files by type, size, or count before they join the active composer list.

Use `onAttachmentCancel` when the host needs to interrupt an in-flight upload. `uinify` does not cancel network work itself; it only exposes the user intent and renders the current blocked or in-progress reason back to the user.

When you transition a composer-owned attachment into a transcript attachment, keep `sourceAttachmentId` stable so the UI can preserve continuity across both phases.
```
