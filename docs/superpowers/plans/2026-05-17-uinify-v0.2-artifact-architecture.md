# Uinify v0.2 Artifact Architecture Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the narrow code-only artifact path with a small, extensible artifact model and renderer architecture that supports metadata, multiple views, default renderers, and stable fallbacks.

**Architecture:** Keep the existing `part.artifact.emitted` runtime event, but widen `UiArtifact` so it can describe identity, title, metadata, and one or more views. In React, route every artifact part through a shared container that handles chrome and view switching, then dispatch the active view through a registry-backed renderer system.

**Tech Stack:** TypeScript, React 18, Vitest, Testing Library, existing runtime and renderer context

Create `src/react/artifact-container.tsx`

```tsx
import { useState } from "react"
import type { UiArtifactPart } from "../model/types"
import { ArtifactBody } from "./artifact-body"

export type ArtifactContainerProps = {
  part: UiArtifactPart
}

export function ArtifactContainer({ part }: ArtifactContainerProps) {
  const { artifact } = part
  const [activeViewId, setActiveViewId] = useState(artifact.defaultViewId ?? artifact.views[0]?.id)
  const activeView = artifact.views.find((view) => view.id === activeViewId) ?? artifact.views[0]

  return (
    <section data-slot="artifact" data-kind={artifact.kind}>
      <header data-slot="artifact-header">
        <strong>{artifact.title ?? artifact.kind}</strong>
        {artifact.metadata
          ? Object.entries(artifact.metadata).map(([key, value]) => <span key={key}>{`${key}: ${value}`}</span>)
          : null}
      </header>
      <div data-slot="artifact-views">
        {artifact.views.map((view) => (
          <button key={view.id} onClick={() => setActiveViewId(view.id)} type="button">
            {view.label}
          </button>
        ))}
      </div>
      {activeView ? <ArtifactBody artifact={artifact} part={part} view={activeView} /> : null}
    </section>
  )
}
```

Create `src/react/artifact-body.tsx`

```tsx
import type { ArtifactRendererProps } from "./renderers"
import { useRenderers } from "./renderers"

export function ArtifactBody({ artifact, part, view }: ArtifactRendererProps) {
  const renderers = useRenderers()
  const renderer = renderers.artifactRenderers?.[artifact.kind]

  if (renderer) {
    return <div data-slot="artifact-body">{renderer({ artifact, part, view })}</div>
  }

  if (typeof view.value === "string") {
    return (
      <div data-slot="artifact-body">
        <pre>{view.value}</pre>
      </div>
    )
  }

  return (
    <div data-slot="artifact-body">
      <pre>{JSON.stringify(view.value, null, 2)}</pre>
    </div>
  )
}
```

```tsx
import { useRenderers } from "./renderers"

export function ArtifactBody({ artifact, part, view }: ArtifactRendererProps) {
  const renderers = useRenderers()
  const renderer = renderers.artifactRenderers?.[artifact.kind]

  if (renderer) {
    return <div data-slot="artifact-body">{renderer({ artifact, part, view })}</div>
  }

  if (renderers.renderArtifactFallback) {
    return (
      <div data-slot="artifact-body">
        {renderers.renderArtifactFallback({ artifact, part, view })}
      </div>
    )
  }

  if (typeof view.value === "string") {
    return <div data-slot="artifact-body"><pre>{view.value}</pre></div>
  }

  return <div data-slot="artifact-body"><pre>{JSON.stringify(view.value, null, 2)}</pre></div>
}
```
