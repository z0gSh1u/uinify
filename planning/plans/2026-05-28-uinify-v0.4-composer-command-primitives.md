# Uinify v0.4 Composer Command Primitives Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace separate slash and mention data models with one unified composer command contract and structured submit selections.

**Architecture:** Keep `LexicalComposer` as the public component, but move command-specific filtering and selection into focused helpers or a plugin when needed. Slash and mention become trigger patterns over the same command array. Submit payloads include selected command records so host code does not rely only on parsing text.

**Tech Stack:** React, Lexical, TypeScript, Testing Library, Vitest, existing composer attachment contracts.

## Public API Decision

`v0.4` is a breaking cleanup, so remove public `slashCommands?: UiComposerChoice[]` and `mentions?: UiComposerChoice[]` from `LexicalComposerProps`. Do not add conversion logic in `LexicalComposer` for those removed props. The only public composer command prop is `commands?: UiComposerCommand[]`.

Keep `src/composer/lexical/plugins/slash-command-plugin.tsx` and `src/composer/lexical/plugins/mention-plugin.tsx` as thin trigger-specific wrappers only if existing internal tests still import those files. Their props must use `UiComposerCommand[]`, not `UiComposerChoice[]`. They are not the recommended public data model and should not appear in new examples.

## Command Model

Use this exact type shape in `src/composer/contracts.ts`:

```ts
export type UiComposerCommandKind =
  | "slash"
  | "mention"
  | "agent"
  | "tool"
  | "skill"
  | "mcp"
  | "custom"

export type UiComposerCommandTrigger = "/" | "@"

export type UiComposerCommand = {
  id: string
  kind: UiComposerCommandKind
  label: string
  insertText: string
  trigger?: UiComposerCommandTrigger
  description?: string
  group?: string
  disabledReason?: string
  metadata?: Record<string, unknown>
}

export type UiComposerCommandSelection = {
  id: string
  kind: UiComposerCommandKind
  label: string
  insertText: string
  trigger: UiComposerCommandTrigger
  range: {
    start: number
    end: number
  }
  description?: string
  group?: string
  metadata?: Record<string, unknown>
}

export type UiComposerValue = {
  text: string
  attachments: UiComposerAttachment[]
  commands: UiComposerCommandSelection[]
}
```

Range is implemented in v0.4. The range records the text offsets occupied by the inserted command token after selection. The old `mentions` submit field is removed instead of retained as an empty array.

Create `src/composer/lexical/plugins/command-plugin.tsx` with:

```tsx
import type { UiComposerCommand, UiComposerCommandTrigger } from "../../contracts"

export type CommandPluginProps = {
  query: string
  commands: UiComposerCommand[]
  onSelect: (command: UiComposerCommand) => void
}

export function CommandPlugin({ query, commands, onSelect }: CommandPluginProps) {
  const matches = filterCommandsForQuery(commands, query)

  if (matches.length === 0) {
    return null
  }

  return (
    <div data-slot="command-menu">
      {matches.map((command) => (
        <button
          aria-describedby={command.disabledReason ? `${command.id}-disabled-reason` : undefined}
          disabled={Boolean(command.disabledReason)}
          key={command.id}
          onClick={() => {
            if (!command.disabledReason) {
              onSelect(command)
            }
          }}
          type="button"
        >
          <span>{command.label}</span>
          <span>{command.kind}</span>
          {command.description ? <small>{command.description}</small> : null}
          {command.disabledReason ? <small id={`${command.id}-disabled-reason`}>{command.disabledReason}</small> : null}
        </button>
      ))}
    </div>
  )
}

export function filterCommandsForQuery(commands: UiComposerCommand[], query: string) {
  const trigger = query[0]

  if (trigger !== "/" && trigger !== "@") {
    return []
  }

  const normalizedQuery = query.slice(1).toLowerCase()

  return commands.filter((command) => {
    if (readCommandTrigger(command) !== trigger) {
      return false
    }

    return command.label.toLowerCase().startsWith(normalizedQuery)
  })
}

export function readCommandTrigger(command: UiComposerCommand): UiComposerCommandTrigger | null {
  if (command.trigger === "/" || command.trigger === "@") {
    return command.trigger
  }

  const firstCharacter = command.insertText.trim()[0]

  return firstCharacter === "/" || firstCharacter === "@" ? firstCharacter : null
}
```

  if (command.disabledReason) {
    return
  }

  const state = getActiveTokenState()

  const trigger = readCommandTrigger(command)

  if (!range || !trigger) {

  setSelectedCommands((current) => [

  ])
}

function createCommandSelection(
  command: UiComposerCommand,
  trigger: "/" | "@",
  start: number,
): UiComposerCommandSelection {
  return {
    id: command.id,
    kind: command.kind,
    label: command.label,
    insertText: command.insertText,
    trigger,
    range: {
      start,
      end: start + command.insertText.length,
    },
    ...(command.description ? { description: command.description } : {}),
    ...(command.group ? { group: command.group } : {}),
    ...(command.metadata ? { metadata: command.metadata } : {}),
  }
}
