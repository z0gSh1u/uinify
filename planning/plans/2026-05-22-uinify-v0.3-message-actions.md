# Uinify v0.3 Message Actions Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a stable transcript action contract with default UI affordances and host-owned callbacks for high-value message and part interactions such as copy, retry, regenerate, reasoning toggle, tool detail toggle, and artifact view actions.

**Architecture:** Keep action state and payload modeling in the React UI layer, not the runtime. Introduce typed action descriptors and structured callback payloads, compute availability from current message and part state, and expose stable action slots so hosts can customize presentation without taking over the entire transcript renderer.

**Tech Stack:** TypeScript, React, Vitest, Testing Library, existing `Message`, `MessagePart`, `ArtifactContainer`, `ReasoningBlock`, and `ToolCallBlock` components

`src/react/actions.ts`

```ts
import type { UiArtifactPart, UiMessage, UiMessagePart } from "../model/types"

export type UiMessageActionId = "copy" | "retry" | "regenerate"
export type UiPartActionId = "copy" | "toggle-reasoning" | "toggle-tool-details" | "open-artifact-view"

export type UiMessageActionDescriptor = {
  id: UiMessageActionId
  label: string
}

export type UiPartActionDescriptor = {
  id: UiPartActionId
  label: string
}

export function getAvailableMessageActions(message: UiMessage): UiMessageActionDescriptor[] {
  if (message.role !== "assistant" || message.state !== "complete") {
    return []
  }

  return [
    { id: "copy", label: "Copy" },
    { id: "retry", label: "Retry" },
    { id: "regenerate", label: "Regenerate" },
  ]
}

export function getAvailablePartActions(part: UiMessagePart): UiPartActionDescriptor[] {
  switch (part.kind) {
    case "reasoning":
      return [
        { id: "toggle-reasoning", label: "Toggle reasoning" },
        { id: "copy", label: "Copy" },
      ]
    case "tool-call":
      return [
        { id: "toggle-tool-details", label: "Toggle tool details" },
        { id: "copy", label: "Copy" },
      ]
    case "artifact":
      return part.artifact.views.length > 1 ? [{ id: "open-artifact-view", label: "Open view" }] : []
    default:
      return part.kind === "text" ? [{ id: "copy", label: "Copy" }] : []
  }
}

export function getArtifactViewPayload(part: UiArtifactPart, viewId: string) {
  return {
    action: "open-artifact-view" as const,
    partId: part.id,
    artifactId: part.artifact.id,
    artifactKind: part.artifact.kind,
    viewId,
  }
}
```

`src/react/chat-root.tsx`

```tsx
type ChatActionHandlers = {
  onMessageAction?: (payload: {
    action: "copy" | "retry" | "regenerate"
    messageId: string
    role: "user" | "assistant" | "tool"
  }) => void
  onPartAction?: (payload: {
    action: "copy" | "toggle-reasoning" | "toggle-tool-details" | "open-artifact-view"
    messageId: string
    partId: string
    partKind: string
    viewId?: string
    artifactId?: string
    artifactKind?: string
  }) => void
}

const ActionHandlersContext = createContext<ChatActionHandlers>({})

export type ChatRootProps = PropsWithChildren<{
  runtime: ReturnType<typeof createChatRuntime>
  renderers?: MessageRendererOverrides
  slotClassNames?: SlotClassNames
}> & ChatActionHandlers

export function useChatActionHandlers() {
  return useContext(ActionHandlersContext)
}
```

`src/react/message-actions.tsx`

```tsx
import type { UiMessage } from "../model/types"
import { getAvailableMessageActions } from "./actions"
import { useChatActionHandlers } from "./chat-root"

export function MessageActions({ message }: { message: UiMessage }) {
  const { onMessageAction } = useChatActionHandlers()
  const actions = getAvailableMessageActions(message)

  if (actions.length === 0) {
    return null
  }

  return (
    <div data-slot="message-actions">
      {actions.map((action) => (
        <button
          key={action.id}
          onClick={() => onMessageAction?.({ action: action.id, messageId: message.id, role: message.role })}
          type="button"
        >
          {action.id === "copy" ? "Copy message" : action.label}
        </button>
      ))}
    </div>
  )
}
```

`src/react/part-actions.tsx`

```tsx
import type { UiMessage, UiMessagePart } from "../model/types"
import { getArtifactViewPayload, getAvailablePartActions } from "./actions"
import { useChatActionHandlers } from "./chat-root"

export function PartActions({ message, part }: { message: UiMessage; part: UiMessagePart }) {
  const { onPartAction } = useChatActionHandlers()
  const actions = getAvailablePartActions(part)

  if (actions.length === 0) {
    return null
  }

  return (
    <div data-slot="part-actions">
      {actions.map((action) => (
        <button
          key={action.id}
          onClick={() => {
            if (action.id === "open-artifact-view" && part.kind === "artifact") {
              const viewId = part.artifact.defaultViewId ?? part.artifact.views[0]?.id

              if (!viewId) {
                return
              }

              onPartAction?.({
                ...getArtifactViewPayload(part, viewId),
                messageId: message.id,
                partKind: part.kind,
              })
              return
            }

            onPartAction?.({
              action: action.id,
              messageId: message.id,
              partId: part.id,
              partKind: part.kind,
            })
          }}
          type="button"
        >
          {action.label}
        </button>
      ))}
    </div>
  )
}
```
