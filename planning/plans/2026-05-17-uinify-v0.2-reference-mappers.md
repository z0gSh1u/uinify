# Uinify v0.2 Reference Mappers Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship example-only reference mappers that show how common backend event streams translate into `UiStreamEvent` without turning `uinify` into a protocol adapter package.

**Architecture:** Keep all protocol-specific code under `examples/adapters/` so the library core stays backend-agnostic. Each mapper should translate one input event into one or more canonical `UiStreamEvent` objects, and tests should prove that mapped output can drive the existing runtime end-to-end.

**Tech Stack:** TypeScript, Vitest, existing `createChatRuntime` runtime API

---

`examples/adapters/openai-like.ts`

```ts
import type { UiArtifact, UiStreamEvent } from "../../src"

export type OpenAiLikeChunk =
  | {
      type: "response.started"
      messageId: string
      role: "user" | "assistant" | "tool"
    }
  | {
      type: "response.output_text.delta"
      messageId: string
      partId: string
      delta: string
    }
  | {
      type: "response.tool.updated"
      messageId: string
      partId: string
      toolName: string
      status: "input-streaming" | "running" | "complete" | "error"
      input?: string
      output?: string
    }
  | {
      type: "response.artifact"
      messageId: string
      partId: string
      artifact: UiArtifact
    }
  | {
      type: "response.completed"
      messageId: string
    }

export function mapOpenAiLikeChunk(chunk: OpenAiLikeChunk): UiStreamEvent[] {
  switch (chunk.type) {
    case "response.started":
      return [{ type: "message.started", messageId: chunk.messageId, role: chunk.role }]

    case "response.output_text.delta":
      return [
        {
          type: "part.text.delta",
          messageId: chunk.messageId,
          partId: chunk.partId,
          delta: chunk.delta,
        },
      ]

    case "response.tool.updated":
      return [
        {
          type: "part.tool.updated",
          messageId: chunk.messageId,
          partId: chunk.partId,
          toolName: chunk.toolName,
          status: chunk.status,
          inputSummary: chunk.input,
          outputSummary: chunk.output,
        },
      ]

    case "response.artifact":
      return [
        {
          type: "part.artifact.emitted",
          messageId: chunk.messageId,
          partId: chunk.partId,
          artifact: chunk.artifact,
        },
      ]

    case "response.completed":
      return [{ type: "message.completed", messageId: chunk.messageId }]
  }
}
```

          content: '{"files":["src/model/types.ts"]}',
        },

import type { UiArtifact, UiStreamEvent } from "../../src"

export type AgentLikeEvent =
  | { type: "agent.message.started"; id: string; role: "user" | "assistant" | "tool" }
  | { type: "agent.message.completed"; id: string }
  | {
      type: "agent.tool.completed"
      id: string
      partId: string
      toolName: string
      inputSummary: string
      outputSummary: string
    }
  | {
      type: "agent.artifact.emitted"
      id: string
      partId: string
      artifact: UiArtifact
    }

export function mapAgentLikeEvent(event: AgentLikeEvent): UiStreamEvent[] {
  switch (event.type) {
    case "agent.message.started":
      return [{ type: "message.started", messageId: event.id, role: event.role }]

    case "agent.message.completed":
      return [{ type: "message.completed", messageId: event.id }]

    case "agent.tool.completed":
      return [
        {
          type: "part.tool.updated",
          messageId: event.id,
          partId: event.partId,
          toolName: event.toolName,
          status: "complete",
          inputSummary: event.inputSummary,
          outputSummary: event.outputSummary,
        },
      ]

    case "agent.artifact.emitted":
      return [
        {
          type: "part.artifact.emitted",
          messageId: event.id,
          partId: event.partId,
          artifact: event.artifact,
        },
      ]
  }
}
