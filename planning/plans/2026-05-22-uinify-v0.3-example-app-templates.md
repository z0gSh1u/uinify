import { useState } from "react"
import { createChatRuntime } from "../../src"
import { ChatRoot, MessageList } from "../../src/react"

export function MinimalAppTemplate() {
  const [runtime] = useState(() => {
    const next = createChatRuntime({ conversationId: "template-minimal" })
    next.dispatch({ type: "message.started", messageId: "m1", role: "assistant" })
    next.dispatch({ type: "part.text.delta", messageId: "m1", partId: "p1", delta: "Minimal app template" })
    next.dispatch({ type: "message.completed", messageId: "m1" })
    return next
  })

  return (
    <section>
      <h2>Minimal app template</h2>
      <ChatRoot runtime={runtime}>
        <MessageList />
      </ChatRoot>
    </section>
  )
}

`examples/templates/adapter-template.tsx`

```tsx
import { useMemo } from "react"
import { createAdapterRunner, createChatRuntime, type UiStreamEvent } from "../../src"
import { ChatRoot, MessageList } from "../../src/react"

type HostEvent = { kind: "begin" | "text" | "end"; id: string; partId?: string; text?: string }

function mapHostEvent(event: HostEvent): UiStreamEvent[] {
  switch (event.kind) {
    case "begin":
      return [{ type: "message.started", messageId: event.id, role: "assistant" }]
    case "text":
      return [{ type: "part.text.delta", messageId: event.id, partId: event.partId!, delta: event.text! }]
    case "end":
      return [{ type: "message.completed", messageId: event.id }]
  }
}

const adaptHostEvent = createAdapterRunner(mapHostEvent)

export function AdapterTemplate() {
  const runtime = useMemo(() => {
    const next = createChatRuntime({ conversationId: "template-adapter" })
    for (const result of [
      adaptHostEvent({ kind: "begin", id: "m1" }),
      adaptHostEvent({ kind: "text", id: "m1", partId: "p1", text: "Adapter integration template" }),
      adaptHostEvent({ kind: "end", id: "m1" }),
    ]) {
      for (const event of result.events) {
        next.dispatch(event)
      }
    }
    return next
  }, [])

  return (
    <section>
      <h2>Adapter integration template</h2>
      <ChatRoot runtime={runtime}>
        <MessageList />
      </ChatRoot>
    </section>
  )
}
```

`examples/templates/upload-template.tsx`

import { useState } from "react"
import type { UiComposerAttachment, UiComposerValue } from "../../src"
import { LexicalComposer } from "../../src/composer/lexical"

export function UploadTemplate() {
  const [attachments, setAttachments] = useState<UiComposerAttachment[]>([])

  function handleSubmit(_value: UiComposerValue) {

  }

  return (
    <section>

      <LexicalComposer
        attachments={attachments}
        onAttachmentsChange={setAttachments}
        onSubmit={handleSubmit}

  )
}

`examples/templates/artifact-template.tsx`

```tsx
import { useMemo } from "react"
import { createChatRuntime } from "../../src"
import { ChatRoot, MessageList, renderDefaultArtifactBody, type MessageRendererOverrides } from "../../src/react"

const renderers: MessageRendererOverrides = {
  artifactRegistry: {
    code: ({ artifact, view }) => (
      <section>
        <h3>{artifact.title ?? "Artifact customization template"}</h3>
        {renderDefaultArtifactBody({ artifact, part: { id: "artifact-template", kind: "artifact", artifact }, view })}
      </section>
    ),
  },
}

export function ArtifactTemplate() {
  const runtime = useMemo(() => {
    const next = createChatRuntime({ conversationId: "template-artifact" })
    next.dispatch({ type: "message.started", messageId: "m1", role: "assistant" })
    next.dispatch({
      type: "part.artifact.emitted",
      messageId: "m1",
      partId: "artifact-1",
      artifact: {
        id: "artifact-1",
        kind: "code",
        title: "Artifact customization template",
        views: [{ id: "source", label: "Source", kind: "source", language: "ts", value: "const customized = true" }],
      },
    })
    next.dispatch({ type: "message.completed", messageId: "m1" })
    return next
  }, [])

  return (
    <section>
      <h2>Artifact customization template</h2>
      <ChatRoot runtime={runtime} renderers={renderers}>
        <MessageList />
      </ChatRoot>
    </section>
  )
}
```
